#!/bin/bash
if ! ls /root &>/dev/null; then
  echo 'Please use "sudo bash" run' && exit 1
fi

if ! grep -q 'Ubuntu' /etc/os-release; then
  read -rp 'Ops, Not Ubuntu Linux, Press any key to continue.' && exit 1
fi

# motd
echo "" | tee /etc/motd &>/dev/null
wget https://raw.githubusercontent.com/yang-lin94/svr-config/refs/heads/main/ubuntu/system.base/motd.sh -qO /etc/profile.d/motd.sh

# alias
wget https://raw.githubusercontent.com/yang-lin94/svr-config/refs/heads/main/ubuntu/system.base/alias.sh -qO /etc/profile.d/alias.sh

# resolv
mkdir /etc/udhcpc &>/dev/null
tee /etc/udhcpc/udhcpc.conf <<EOF
RESOLV_CONF="no"
EOF

tee /etc/resolv.conf <<EOF
nameserver 168.95.1.1
nameserver 8.8.8.8
EOF
